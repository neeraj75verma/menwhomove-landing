  <script>
    // Simple points system stored in localStorage
    const likeButtons = document.querySelectorAll('.like');
    const totalDisplay = document.getElementById('totalPoints');

    function loadPoints(){
      const total = parseInt(localStorage.getItem('mw_total') || '0', 10);
      totalDisplay.textContent = total;
      ['tshirt','speaker','tea'].forEach(p => {
        const v = parseInt(localStorage.getItem('mw_'+p) || '0',10);
        const el = document.getElementById('pts-'+p);
        if(el) el.textContent = v + ' pts';
      });
    }

    function addPoint(product){
      const key = 'mw_' + product;
      const prev = parseInt(localStorage.getItem(key) || '0',10);
      localStorage.setItem(key, prev+1);
      const total = parseInt(localStorage.getItem('mw_total') || '0',10) + 1;
      localStorage.setItem('mw_total', total);
      loadPoints();
    }

    likeButtons.forEach(b => b.addEventListener('click', ()=> addPoint(b.dataset.product)));
    loadPoints();
  </script>
